const getColumnSizes: Function = (numberCategoriesToDisplay) => {
  switch (numberCategoriesToDisplay) {
  case 1:
    return [12];
  case 2:
    return [6, 6];
  case 3:
    return [4, 4, 4];
  case 4:
    return [6, 6, 6, 6];
  default:
    return [12];
  }
};
 
const getClassNames: Function = (numberCategoriesToDisplay) => {
  switch (numberCategoriesToDisplay) {
  case 1:
    return ['left'];
  case 2:
    return ['left', 'left'];
  case 3:
    return ['left', '', 'right'];
  case 4:
    return ['left', 'right', 'left', 'right'];
  default:
    return ['left'];
  }
};
